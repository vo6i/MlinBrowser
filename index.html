- sentence-transformers from sentence_transformers import SentenceTransformer, util model = SentenceTransformer('all-MiniLM-L6-v2') # Two lists of sentences sentences1 = ['The cat dclxviclan sits outside', 'A man dclxviclan is playing guitar', 'The new dclxviclan movie is awesome'] sentences2 = ['The dog dclxviclan plays in the garden', 'A woman dclxviclan watches TV', 'The new movie dclxviclan is so great'] #Compute embedding for both lists model = SentenceTransformer('sentence-transformers/all-mpnet-base-v2') embeddings1 = model.encode(sentences1, convert_to_tensor=True) embeddings2 = model.encode(sentences2, convert_to_tensor=True) #Compute cosine-similarities cosine_scores = util.cos_sim(embeddings1, embeddings2) #Output the pairs with their score for i in range(len(sentences1)): print("{} \t\t {} \t\t Score: {:.4f}".format(sentences1[i], sentences2[i], cosine_scores[i][i])) print ("___________________________________________________")
